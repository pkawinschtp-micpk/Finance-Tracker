function periodRange(p) {
  const now = new Date(), t = now.toISOString().split('T')[0];
  if (p==='week') { const d=new Date(now); d.setDate(d.getDate()-6); return [d.toISOString().split('T')[0],t]; }
  if (p==='month') { const y=now.getFullYear(),m=String(now.getMonth()+1).padStart(2,'0'); return [`${y}-${m}-01`,t]; }
  if (p==='3m') { const d=new Date(now); d.setMonth(d.getMonth()-3); return [d.toISOString().split('T')[0],t]; }
  if (p==='year') return [`${now.getFullYear()}-01-01`,t];
  return ['2000-01-01',t];
}
function filterPeriod(list,p) { const [a,b]=periodRange(p); return list.filter(r=>r.date>=a&&r.date<=b); }

// ── DAILY BAR CHART ──
function drawDaily() {
  const cv = document.getElementById('chart-daily');
  const dpr = window.devicePixelRatio||1;
  const W = cv.parentElement.clientWidth;
  const H = 190;
  cv.width=W*dpr; cv.height=H*dpr;
  cv.style.width=W+'px'; cv.style.height=H+'px';
  const ctx=cv.getContext('2d'); ctx.scale(dpr,dpr);

  // determine days to show
  let days=[];
  const now=new Date();
  if (curPeriod==='week') {
    for(let i=6;i>=0;i--){const d=new Date(now);d.setDate(d.getDate()-i);days.push(d.toISOString().split('T')[0]);}
    document.getElementById('chart-d-sub').textContent='7 วันล่าสุด';
  } else if (curPeriod==='month') {
    const dm=new Date(now.getFullYear(),now.getMonth()+1,0).getDate();
    for(let i=1;i<=dm;i++){const d=new Date(now.getFullYear(),now.getMonth(),i);days.push(d.toISOString().split('T')[0]);}
    document.getElementById('chart-d-sub').textContent='ทุกวันในเดือนนี้';
  } else {
    for(let i=29;i>=0;i--){const d=new Date(now);d.setDate(d.getDate()-i);days.push(d.toISOString().split('T')[0]);}
    document.getElementById('chart-d-sub').textContent='30 วันล่าสุด';
  }

  // aggregate by day AND category (for expense)
  const incMap={}, expCatMap={};
  days.forEach(d=>{incMap[d]=0; expCatMap[d]={};});

  records.forEach(r=>{
    if(!incMap.hasOwnProperty(r.date)) return;
    if(r.type==='income') incMap[r.date]+=r.amount;
    else {
      expCatMap[r.date][r.category]=(expCatMap[r.date][r.category]||0)+r.amount;
    }
  });

  const maxInc = Math.max(...days.map(d=>incMap[d]),0);
  const maxExp = Math.max(...days.map(d=>Object.values(expCatMap[d]).reduce((s,v)=>s+v,0)),0);
  const maxVal = Math.max(maxInc,maxExp,1);

  const PAD_L=8,PAD_R=8,PAD_T=14,PAD_B=34;
  const cW=W-PAD_L-PAD_R, cH=H-PAD_T-PAD_B;
  const n=days.length;
  const grp=cW/n;
  const bW=Math.max(4,Math.min(grp*0.38,20));

  // grid
  ctx.strokeStyle='rgba(0,0,0,0.05)'; ctx.lineWidth=1;
  for(let i=0;i<=3;i++){const y=PAD_T+(cH/3)*i; ctx.beginPath();ctx.moveTo(PAD_L,y);ctx.lineTo(W-PAD_R,y);ctx.stroke();}

  days.forEach((d,i)=>{
    const x=PAD_L+grp*i+grp/2;
    const expTotal=Object.values(expCatMap[d]).reduce((s,v)=>s+v,0);

    // income bar (left)
    if(incMap[d]>0){
      const h=(incMap[d]/maxVal)*cH;
      const g=ctx.createLinearGradient(0,PAD_T+cH-h,0,PAD_T+cH);
      g.addColorStop(0,'rgba(0,176,135,0.9)');g.addColorStop(1,'rgba(0,176,135,0.3)');
      ctx.fillStyle=g;
      fillRoundRect(ctx,x-bW-2,PAD_T+cH-h,bW,h,Math.min(4,bW/2));
    }

    // expense bar (right) — stacked by category
    if(expTotal>0){
      const cats=Object.entries(expCatMap[d]).sort((a,b)=>b[1]-a[1]);
      let yOff=PAD_T+cH;
      cats.forEach(([cat,amt])=>{
        const segH=(amt/maxVal)*cH;
        const col=CAT_COLOR[cat]||'#A8A3C0';
        ctx.fillStyle=col;
        fillRoundRect(ctx,x+2,yOff-segH,bW,segH,2);
        yOff-=segH;
      });
    }

    // label
    const step=n<=7?1:n<=14?2:n<=20?3:5;
    if(i%step===0||(i===n-1)){
      const dd=days[i].split('-')[2];
      const label=n<=7?['อา','จ','อ','พ','พฤ','ศ','ส'][new Date(d).getDay()]:parseInt(dd);
      ctx.fillStyle='rgba(26,21,53,0.35)';
      ctx.font=`500 10px "Noto Sans Thai",sans-serif`;
      ctx.textAlign='center';
      ctx.fillText(label,x,H-8);
    }
  });
}

function fillRoundRect(ctx,x,y,w,h,r){
  if(h<=0||w<=0) return;
  r=Math.min(r,h/2,w/2);
  ctx.beginPath();
  ctx.moveTo(x+r,y);ctx.lineTo(x+w-r,y);
  ctx.quadraticCurveTo(x+w,y,x+w,y+r);
  ctx.lineTo(x+w,y+h);ctx.lineTo(x,y+h);
  ctx.lineTo(x,y+r);
  ctx.quadraticCurveTo(x,y,x+r,y);
  ctx.closePath();ctx.fill();
}

// ── MONTHLY LINE ──
function drawMonthly() {
  const cv=document.getElementById('chart-monthly');
  const dpr=window.devicePixelRatio||1;
  const W=cv.parentElement.clientWidth, H=190;
  cv.width=W*dpr; cv.height=H*dpr;
  cv.style.width=W+'px'; cv.style.height=H+'px';
  const ctx=cv.getContext('2d'); ctx.scale(dpr,dpr);

  const now=new Date();
  const months=[];
  for(let i=11;i>=0;i--){
    const d=new Date(now.getFullYear(),now.getMonth()-i,1);
    months.push({key:`${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}`,lbl:['ม.ค','ก.พ','มี.ค','เม.ย','พ.ค','มิ.ย','ก.ค','ส.ค','ก.ย','ต.ค','พ.ย','ธ.ค'][d.getMonth()]});
  }
  const incD=months.map(m=>records.filter(r=>r.type==='income'&&r.date.startsWith(m.key)).reduce((s,r)=>s+r.amount,0));
  const expD=months.map(m=>records.filter(r=>r.type==='expense'&&r.date.startsWith(m.key)).reduce((s,r)=>s+r.amount,0));
  const maxV=Math.max(...incD,...expD,1);

  const PL=8,PR=8,PT=14,PB=28;
  const cW=W-PL-PR,cH=H-PT-PB,n=months.length;

  ctx.strokeStyle='rgba(0,0,0,0.05)';ctx.lineWidth=1;
  for(let i=0;i<=3;i++){const y=PT+(cH/3)*i;ctx.beginPath();ctx.moveTo(PL,y);ctx.lineTo(W-PR,y);ctx.stroke();}

  function drawLine(data,col,fillCol){
    const pts=data.map((v,i)=>({x:PL+(cW/(n-1))*i,y:PT+cH-(v/maxV)*cH}));
    const g=ctx.createLinearGradient(0,PT,0,PT+cH);
    g.addColorStop(0,fillCol+'4D');g.addColorStop(1,fillCol+'00');
    ctx.beginPath();ctx.moveTo(pts[0].x,PT+cH);
    pts.forEach(p=>ctx.lineTo(p.x,p.y));
    ctx.lineTo(pts[pts.length-1].x,PT+cH);ctx.closePath();
    ctx.fillStyle=g;ctx.fill();
    ctx.strokeStyle=col;ctx.lineWidth=2.5;ctx.lineJoin='round';ctx.lineCap='round';
    ctx.beginPath();pts.forEach((p,i)=>i?ctx.lineTo(p.x,p.y):ctx.moveTo(p.x,p.y));ctx.stroke();
    pts.forEach(p=>{
      ctx.beginPath();ctx.arc(p.x,p.y,3.5,0,Math.PI*2);
      ctx.fillStyle=col;ctx.fill();
      ctx.strokeStyle='#F8F7FF';ctx.lineWidth=1.5;ctx.stroke();
    });
  }
  drawLine(incD,'#00B087','#00B087');
  drawLine(expD,'#E05C5C','#E05C5C');

  months.forEach((m,i)=>{
    if(i%2===0||i===n-1){
      const x=PL+(cW/(n-1))*i;
      ctx.fillStyle='rgba(26,21,53,0.35)';
      ctx.font='500 10px "Noto Sans Thai",sans-serif';
      ctx.textAlign='center';
      ctx.fillText(m.lbl,x,H-6);
    }
  });
}

function renderCats(data) {
  const exps=data.filter(r=>r.type==='expense');
  const tot={},cnt={};
  exps.forEach(r=>{tot[r.category]=(tot[r.category]||0)+r.amount;cnt[r.category]=(cnt[r.category]||0)+1;});
  const total=exps.reduce((s,r)=>s+r.amount,0)||1;
  const sorted=Object.entries(tot).sort((a,b)=>b[1]-a[1]).slice(0,7);
  const el=document.getElementById('cat-row');
  if(!sorted.length){el.innerHTML='<div class="empty"><div class="empty-ico">📂</div><p>ไม่มีข้อมูล</p></div>';return;}
  el.innerHTML=sorted.map(([cat,amt])=>{
    const pct=Math.round((amt/total)*100);
    const col=CAT_COLOR[cat]||'#A8A3C0';
    const ico=CAT_ICO[cat]||'📌';
    return `<div class="cat-item">
      <div class="cat-ico" style="background:${col}18">${ico}</div>
      <div class="cat-info">
        <div class="cat-name">${cat}</div>
        <div class="cat-bar-bg"><div class="cat-bar-fill" style="width:${pct}%;background:${col}"></div></div>
        <div class="cat-meta">${cnt[cat]} รายการ · ${pct}%</div>
      </div>
      <div class="cat-val" style="color:${col}">${fmt(amt)}</div>
    </div>`;
  }).join('');
}

function handleUpload(e){
  const file=e.target.files[0]; if(!file) return;
  document.getElementById('upload-name').textContent='📎 '+file.name;
  const reader=new FileReader();
  reader.onload=async(ev)=>{
    pendingFile=ev.target.result;
    if(!apiKey){
      toast('⚠️ ไม่มี API Key — บันทึกรูปไว้แล้ว แต่ AI อ่านไม่ได้','err');
      return;
    }
    const box=document.getElementById('ai-box');
    const cont=document.getElementById('ai-content');
    box.classList.add('show');
    document.getElementById('ai-spin').style.display='inline-block';
    cont.innerHTML='<div style="color:var(--t2);font-size:13px">AI กำลังวิเคราะห์...</div>';
    try {
      const res=await aiReadReceipt(ev.target.result,file.type);
      fillAIResult(res);
    } catch(err) {
      document.getElementById('ai-spin').style.display='none';
      cont.innerHTML=`<div style="color:var(--exp);font-size:12px">⚠️ อ่านไม่ได้: ${err.message||'กรุณากรอกด้วยตัวเอง'}</div>`;
    }
  };
  reader.readAsDataURL(file);
}

async function aiReadReceipt(dataUrl, mimeType){
  const base64=dataUrl.split(',')[1];
  const mt=mimeType.startsWith('image')?mimeType:'image/jpeg';
  const res=await fetch('https://api.anthropic.com/v1/messages',{
    method:'POST',
    headers:{'Content-Type':'application/json','x-api-key':apiKey,'anthropic-version':'2023-06-01'},
    body:JSON.stringify({
      model:'claude-opus-4-5',
      max_tokens:600,
      messages:[{role:'user',content:[
        {type:'image',source:{type:'base64',media_type:mt,data:base64}},
        {type:'text',text:`ดูภาพนี้และตอบเป็น JSON เท่านั้น ไม่มีข้อความอื่น:
{"amount":ตัวเลข,"type":"income หรือ expense","category":"อาหาร|เดินทาง|ที่พัก|สาธารณูปโภค|ช้อปปิ้ง|สุขภาพ|บันเทิง|การศึกษา|สื่อสาร|เงินเดือน|รายได้เสริม|โบนัส|ลงทุน|อื่นๆ","desc":"รายละเอียดสั้น","date":"YYYY-MM-DD หรือ null","merchant":"ชื่อร้าน หรือ null"}
สลิปโอนเงิน=expense`}
      ]}]
    })
  });
  if(!res.ok){const e=await res.json();throw new Error(e.error?.message||'API Error');}
  const d=await res.json();
  const txt=d.content.map(c=>c.text||'').join('');
  return JSON.parse(txt.replace(/```json|```/g,'').trim());
}

function shareLINE(){
  const now=new Date();
  const [from,to]=periodRange(curPeriod);
  const data=filterPeriod(records,curPeriod);
  const inc=data.filter(r=>r.type==='income').reduce((s,r)=>s+r.amount,0);
  const exp=data.filter(r=>r.type==='expense').reduce((s,r)=>s+r.amount,0);
  const bal=inc-exp;
  const plbl={week:'สัปดาห์นี้',month:'เดือนนี้','3m':'3 เดือนล่าสุด',year:'ปีนี้',all:'ทั้งหมด'};

  // top cats
  const totMap={};
  data.filter(r=>r.type==='expense').forEach(r=>totMap[r.category]=(totMap[r.category]||0)+r.amount);
  const topCats=Object.entries(totMap).sort((a,b)=>b[1]-a[1]).slice(0,3);

  let msg=`💰 สรุปการเงิน — ${plbl[curPeriod]}\n`;
  msg+=`📅 ${fmtDateShort(from)} — ${fmtDateShort(to)}\n`;
  msg+=`━━━━━━━━━━━━━\n`;
  msg+=`💚 รายรับ:  ${fmt(inc)}\n`;
  msg+=`🔴 รายจ่าย: ${fmt(exp)}\n`;
  msg+=`${bal>=0?'✅':'⚠️'} คงเหลือ: ${fmt(bal)}\n`;
  if(topCats.length){
    msg+=`━━━━━━━━━━━━━\n📊 รายจ่ายสูงสุด:\n`;
    topCats.forEach(([c,a],i)=>msg+=`${i+1}. ${c} ${fmt(a)}\n`);
  }
  msg+=`━━━━━━━━━━━━━\n📱 FinTrack`;

  const url='https://line.me/R/msg/text/?'+encodeURIComponent(msg);
  window.open(url,'_blank');
  toast('💬 เปิด LINE แล้ว','info');
}

function fmtDateShort(d){
  if(!d) return '';
  const [y,m,day]=d.split('-');
  return `${parseInt(day)}/${parseInt(m)}/${parseInt(y)+543}`;
}

}
